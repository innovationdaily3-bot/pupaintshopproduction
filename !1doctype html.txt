  <script>
    const defaultConfig = {
      dashboard_title: "Production Dashboard",
      company_name: "Manufacturing Analytics"
    };

    let allData = [];
    let productionChart = null;
    let defectChart = null;

    const dataHandler = {
      onDataChanged(data) {
        allData = data;
        updateDashboard();
      }
    };

    async function onConfigChange(config) {
      document.getElementById('dashboardTitle').textContent = config.dashboard_title || defaultConfig.dashboard_title;
      document.getElementById('companyName').textContent = config.company_name || defaultConfig.company_name;
    }

    function mapToCapabilities(config) {
      return {
        recolorables: [],
        borderables: [],
        fontEditable: undefined,
        fontSizeable: undefined
      };
    }

    function mapToEditPanelValues(config) {
      return new Map([
        ['dashboard_title', config.dashboard_title || defaultConfig.dashboard_title],
        ['company_name', config.company_name || defaultConfig.company_name]
      ]);
    }

    async function init() {
      if (window.dataSdk) {
        const result = await window.dataSdk.init(dataHandler);
        if (!result.isOk) {
          console.error('Failed to initialize data SDK');
        }
      }

      if (window.elementSdk) {
        window.elementSdk.init({
          defaultConfig,
          onConfigChange,
          mapToCapabilities,
          mapToEditPanelValues
        });
      }

      setupEventListeners();
      updateDashboard();
    }

    function setupEventListeners() {
      document.getElementById('addDataBtn').addEventListener('click', () => {
        document.getElementById('dataModal').classList.add('active');
        document.getElementById('date').valueAsDate = new Date();
      });

      document.getElementById('closeModal').addEventListener('click', closeModal);
      document.getElementById('cancelBtn').addEventListener('click', closeModal);

      document.getElementById('dataForm').addEventListener('submit', handleSubmit);

      document.getElementById('downloadBtn').addEventListener('click', downloadExcel);
    }

    function closeModal() {
      document.getElementById('dataModal').classList.remove('active');
      document.getElementById('dataForm').reset();
    }

    async function handleSubmit(e) {
      e.preventDefault();

      if (allData.length >= 999) {
        showToast('Maximum limit of 999 records reached. Please delete some records first.');
        return;
      }

      const submitBtn = document.getElementById('submitBtn');
      const submitText = document.getElementById('submitText');
      
      submitBtn.disabled = true;
      submitText.innerHTML = '<span class="loading"></span>';

      const formData = {
        date: document.getElementById('date').value,
        shift: document.getElementById('shift').value,
        partName: document.getElementById('partName').value,
        colour: document.getElementById('colour').value,
        runHr: parseInt(document.getElementById('runHr').value),
        targetProd: parseInt(document.getElementById('targetProd').value) || 0,
        totalLoading: parseInt(document.getElementById('totalLoading').value),
        okQty: parseInt(document.getElementById('okQty').value),
        reworkQty: parseInt(document.getElementById('reworkQty').value),
        reworkPercent: 0,
        buffing: parseInt(document.getElementById('buffing').value) || 0,
        finalScrap: parseInt(document.getElementById('finalScrap').value) || 0,
        dust: parseInt(document.getElementById('dust').value) || 0,
        rd: parseInt(document.getElementById('rd').value) || 0,
        scratch: parseInt(document.getElementById('scratch').value) || 0,
        cissing: parseInt(document.getElementById('cissing').value) || 0,
        wm: parseInt(document.getElementById('wm').value) || 0,
        dry: 0,
        orangePeel: 0,
        damage: 0,
        uncovered: 0,
        noLaq: 0,
        touchUp: 0,
        colourNG: 0,
        paintCrack: 0,
        ribBroken: 0,
        screwMiss: 0,
        glossNG: 0,
        patchMark: 0,
        mouldingIssue: 0,
        waterShrinkage: 0,
        spitting: 0,
        scratchOther: 0,
        others: 0
      };

      const total = formData.okQty + formData.reworkQty;
      formData.reworkPercent = total > 0 ? parseFloat(((formData.reworkQty / total) * 100).toFixed(1)) : 0;

      const result = await window.dataSdk.create(formData);

      submitBtn.disabled = false;
      submitText.textContent = 'Save Data';

      if (result.isOk) {
        showToast('Production data saved successfully!');
        closeModal();
      } else {
        showToast('Failed to save data. Please try again.');
      }
    }

    async function deleteRecord(record) {
      const btn = event.target;
      btn.disabled = true;
      btn.textContent = 'Deleting...';

      const result = await window.dataSdk.delete(record);

      if (result.isOk) {
        showToast('Record deleted successfully!');
      } else {
        showToast('Failed to delete record. Please try again.');
        btn.disabled = false;
        btn.textContent = 'Delete';
      }
    }

    function updateDashboard() {
      updateStats();
      updateCharts();
      updateTable();
    }

    function updateStats() {
      const totalRecords = allData.length;
      const totalOkQty = allData.reduce((sum, item) => sum + (item.okQty || 0), 0);
      const totalRework = allData.reduce((sum, item) => sum + (item.reworkQty || 0), 0);
      const avgRework = allData.length > 0 
        ? (allData.reduce((sum, item) => sum + (item.reworkPercent || 0), 0) / allData.length).toFixed(1)
        : 0;

      document.getElementById('totalRecords').textContent = totalRecords;
      document.getElementById('totalOkQty').textContent = totalOkQty.toLocaleString();
      document.getElementById('totalRework').textContent = totalRework.toLocaleString();
      document.getElementById('avgRework').textContent = avgRework + '%';
    }

    function updateCharts() {
      const recentData = allData.slice(-10);
      
      const ctx1 = document.getElementById('productionChart');
      if (productionChart) {
        productionChart.destroy();
      }
      
      productionChart = new Chart(ctx1, {
        type: 'bar',
        data: {
          labels: recentData.map(d => d.date),
          datasets: [
            {
              label: 'OK Quantity',
              data: recentData.map(d => d.okQty),
              backgroundColor: 'rgba(102, 126, 234, 0.8)',
              borderColor: 'rgba(102, 126, 234, 1)',
              borderWidth: 1
            },
            {
              label: 'Rework Quantity',
              data: recentData.map(d => d.reworkQty),
              backgroundColor: 'rgba(252, 129, 129, 0.8)',
              borderColor: 'rgba(252, 129, 129, 1)',
              borderWidth: 1
            }
          ]
        },
        options: {
          responsive: true,
          maintainAspectRatio: true,
          scales: {
            y: {
              beginAtZero: true
            }
          }
        }
      });

      const defectTypes = ['dust', 'rd', 'scratch', 'cissing', 'wm'];
      const defectCounts = defectTypes.map(type => 
        allData.reduce((sum, item) => sum + (item[type] || 0), 0)
      );

      const ctx2 = document.getElementById('defectChart');
      if (defectChart) {
        defectChart.destroy();
      }
      
      defectChart = new Chart(ctx2, {
        type: 'pie',
        data: {
          labels: ['Dust', 'RD', 'Scratch', 'Cissing', 'WM'],
          datasets: [{
            data: defectCounts,
            backgroundColor: [
              'rgba(102, 126, 234, 0.8)',
              'rgba(118, 75, 162, 0.8)',
              'rgba(72, 187, 120, 0.8)',
              'rgba(237, 137, 54, 0.8)',
              'rgba(252, 129, 129, 0.8)'
            ]
          }]
        },
        options: {
          responsive: true,
          maintainAspectRatio: true
        }
      });
    }

    function updateTable() {
      const tableContent = document.getElementById('tableContent');

      if (allData.length === 0) {
        tableContent.innerHTML = `
          <div class="empty-state">
            <h3>No Data Yet</h3>
            <p>Click "Add Production Data" to start tracking your production metrics</p>
          </div>
        `;
        return;
      }

      const sortedData = [...allData].sort((a, b) => new Date(b.date) - new Date(a.date));

      let html = `
        <table class="data-table">
          <thead>
            <tr>
              <th>Date</th>
              <th>Shift</th>
              <th>Part Name</th>
              <th>Colour</th>
              <th>Run Hr</th>
              <th>Total Loading</th>
              <th>OK Qty</th>
              <th>Rework Qty</th>
              <th>Rework %</th>
              <th>Actions</th>
            </tr>
          </thead>
          <tbody>
      `;

      sortedData.forEach(record => {
        html += `
          <tr>
            <td>${record.date}</td>
            <td>${record.shift}</td>
            <td>${record.partName}</td>
            <td>${record.colour}</td>
            <td>${record.runHr}</td>
            <td>${record.totalLoading}</td>
            <td>${record.okQty}</td>
            <td>${record.reworkQty}</td>
            <td>${record.reworkPercent}%</td>
            <td>
              <button class="delete-btn" onclick="deleteRecord(${JSON.stringify(record).replace(/"/g, '&quot;')})">
                Delete
              </button>
            </td>
          </tr>
        `;
      });

      html += `
          </tbody>
        </table>
      `;

      tableContent.innerHTML = html;
    }

    function downloadExcel() {
      if (allData.length === 0) {
        showToast('No data to download');
        return;
      }

      const exportData = allData.map((record, index) => ({
        'S.NO': index + 1,
        'DATE': record.date,
        'SHIFT': record.shift,
        'Part Name': record.partName,
        'Colour': record.colour,
        'Run Hr': record.runHr,
        'Target Prod.': record.targetProd,
        'Total loading': record.totalLoading,
        'OK Qty': record.okQty,
        'Rework qty': record.reworkQty,
        'Rework %': record.reworkPercent + '%',
        'Buffing': record.buffing,
        'Final Scrap': record.finalScrap,
        'Dust': record.dust,
        'RD': record.rd,
        'Scratch': record.scratch,
        'Cissing': record.cissing,
        'WM': record.wm
      }));

      const ws = XLSX.utils.json_to_sheet(exportData);
      const wb = XLSX.utils.book_new();
      XLSX.utils.book_append_sheet(wb, ws, 'Production Data');
      
      const fileName = `production_data_${new Date().toISOString().split('T')[0]}.xlsx`;
      XLSX.writeFile(wb, fileName);
      
      showToast('Excel file downloaded successfully!');
    }

    function showToast(message) {
      const toast = document.getElementById('toast');
      toast.textContent = message;
      toast.classList.add('show');
      setTimeout(() => {
        toast.classList.remove('show');
      }, 3000);
    }

    window.deleteRecord = deleteRecord;
